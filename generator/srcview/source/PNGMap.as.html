package
{
    import flash.display.Bitmap;
    import flash.display.Graphics;
    import flash.display.Loader;
    import flash.display.Sprite;
    import flash.display.StageAlign;
    import flash.display.StageScaleMode;
    import flash.events.Event;
    import flash.events.MouseEvent;
    import flash.text.TextField;
    import flash.text.TextFieldAutoSize;
    import flash.text.TextFormat;
    
    public class PNGMap extends Sprite
    {
        [Embed("bretagne.png")]
        private var bitmapTexture:Class;
        
        private var image:ImageReference;
        private var border:uint = 20;
        
        private var source:Bitmap;
        private var preview:Sprite;

        private var bg1:Sprite;
        private var bg2:Sprite;
        

        private var txtHead:TextField;
        private var txtTitle1:TextField;
        private var txtTitle2:TextField;
        private var txtData:TextField;
        private var txtRoll:TextField;
        private var txtDataTitle:TextField;
        
        private var data:Object = {}; //{134:[[173,177]],135:[[159,161],[173,177],[178,184]]}
        
        public function PNGMap()
        {
            stage.scaleMode = StageScaleMode.NO_SCALE;
            stage.align = StageAlign.TOP_LEFT;
            
            image = new ImageReference();
            image.addEventListener(Event.COMPLETE, onImageData);
            
            txtHead = new TextField();
            txtHead.autoSize = TextFieldAutoSize.LEFT;
            txtHead.defaultTextFormat = new TextFormat("Verdana", 15, 0x333333, true);
            txtHead.x = border;
            txtHead.y = border;
            txtHead.text = "PNG MAP Generator";
            addChild(txtHead);
            
            txtTitle1 = new TextField();
            txtTitle1.defaultTextFormat = new TextFormat("Verdana", 11, 0x333333);
            txtTitle1.autoSize = TextFieldAutoSize.LEFT;
            txtTitle1.x = border;
            txtTitle1.y = txtHead.y+txtHead.height+border;
            txtTitle1.text = "Original PNG (click to change)";
            addChild(txtTitle1);
            
            bg1 = new Sprite();
            bg1.x = border;
            bg1.y = txtTitle1.y + txtTitle1.height+border*0.5;
            bg1.graphics.beginFill(0xCCCCCC);
            bg1.graphics.drawRect(0, 0, 400, 400);
            bg1.graphics.endFill();
            bg1.addEventListener(MouseEvent.CLICK, onImageClick);
            addChild(bg1);
            
            bg2 = new Sprite();
            bg2.x = bg1.x+bg1.width+border;
            bg2.y = bg1.y;
            bg2.graphics.beginFill(0xCCCCCC);
            bg2.graphics.drawRect(0, 0, 400, 400);
            bg2.graphics.endFill();
            bg2.addEventListener(MouseEvent.CLICK, onImageClick);
            addChild(bg2);
            
            txtTitle2 = new TextField();
            txtTitle2.defaultTextFormat = new TextFormat("Verdana", 11, 0x333333);
            txtTitle2.autoSize = TextFieldAutoSize.LEFT;
            txtTitle2.wordWrap = true;
            txtTitle2.x = bg2.x;
            txtTitle2.y = txtTitle1.y;
            txtTitle2.text = "MAP Preview";
            addChild(txtTitle2);
            
            source = new Bitmap();
            source.x = bg1.x;
            source.y = bg1.y;
            addChild(source);
            
            preview = new Sprite();
            preview.x = bg2.x;
            preview.y = bg2.y;
            addChild(preview);
            
            txtRoll = new TextField();
            txtRoll.autoSize = TextFieldAutoSize.LEFT;
            txtRoll.defaultTextFormat = new TextFormat("Verdana", 11, 0x333333, true);
            txtRoll.height = 20;
            txtRoll.text = "Mouse over ?";
            txtRoll.x = bg1.x,
            txtRoll.y = bg1.y+bg1.height+border;
            addChild(txtRoll);
            
            txtDataTitle = new TextField();
            txtDataTitle.autoSize = TextFieldAutoSize.LEFT;
            txtDataTitle.defaultTextFormat = new TextFormat("Verdana", 11, 0x333333, true);
            txtDataTitle.height = 20;
            txtDataTitle.text = "Javascript data :";
            txtDataTitle.x = bg2.x,
            txtDataTitle.y = bg1.y+bg1.height+border;
            addChild(txtDataTitle);
            
            txtData = new TextField();
            txtData.defaultTextFormat = new TextFormat("Verdana", 11, 0x333333);
            txtData.multiline = true;
            txtData.wordWrap = true;
            txtData.border = true;
            txtData.x = bg2.x;
            txtData.y = txtDataTitle.y+txtDataTitle.height;
            txtData.width =  bg2.width;
            txtData.height = 50;
            addChild(txtData);
            
            source.bitmapData = (new bitmapTexture() as Bitmap).bitmapData;
            
            onImageReady();
        }
        
        protected function onImageData(event:Event):void{
            source.bitmapData = image.data;
            onImageReady();
        }
        
        protected function onImageReady():void {
            map();
            txtData.text = "var data = "+getMapString();
            drawMap();
            bg1.addEventListener(MouseEvent.MOUSE_MOVE, update);
        }
        
        protected function onImageClick(event:Event):void{
            image.browse();
        }
        
        protected function map():void {
            data = {};
            
            var col:uint = source.width;
            var row:uint = source.height;
            var pix:uint;
            
            var a:Array;
            
            for(var i:uint=0; i<row; i++){
                
                var start:int = -1;
                var end:int = -1;
                a = [-1, -1];
                
                for(var j:uint=0; j<col; j++){
                    pix = source.bitmapData.getPixel32(j, i);
                    
                    if(pix != 0 && a[0] == -1){
                        a[0] = j;
                    }else if(pix == 0 && a[0] != -1 && a[1] == -1){
                        a[1] = j;
                        
                        if(data[i] == null)
                            data[i] = [];
                        data[i].push(a);
                        a = [-1, -1];
                    }
                }
            }
        }
        
        protected function drawMap():void {
            preview.graphics.clear();
            for(var i:String in data){
                var n:uint = data[i].length;
                for(var j:uint=0; j<n; j++){
                    if(data[i][j][0] != -1 && data[i][j][1] != -1){
                        preview.graphics.beginFill(0xFF0000);
                        preview.graphics.drawRect(data[i][j][0], uint(i), data[i][j][1]-data[i][j][0], 1);
                        preview.graphics.endFill();
                    }
                }
            }
        }
        
        protected function getMapString():String {
            
            var s:String = '{';
            for(var i:String in data){
                if(s != "{")
                    s+=",";
                var n:uint = data[i].length;
                s += i+":[";
                for(var j:uint=0; j<n; j++){
                    if(j>0)
                        s+=",";
                    s += "["+data[i][j][0]+","+data[i][j][1]+"]";
                }
                s+="]";
            }
            s+= "}";
            return s;
        }
        
        protected function update(e:MouseEvent):void {
            e.stopImmediatePropagation();
            var r:uint = e.localY;
            var c:uint = e.localX;

            var n:uint;
            var b:Boolean = false;
                
            if(r <= source.height && r <= source.width && data[r] != null)
            {
                n = data[r].length;
                for(var i:uint = 0; i<n; i++){
                    if(c > data[r][i][0] && c < data[r][i][1])
                        b = true;
                }
                
                txtRoll.text = "Mouse over ? " + b ;
            }
        }
    }
}
